rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function signedIn() {
      return request.auth != null;
    }

    function isSuperAdmin() {
      return signedIn()
        && request.auth.token.email != null
        && request.auth.token.email in get(/databases/$(database)/documents/superback_config/main).data.adminEmails;
    }

    match /superback_config/{docId} {
      allow read: if isSuperAdmin();
      allow write: if false;
    }

    match /tenants_public/{tenantId} {
      allow read, write: if isSuperAdmin();
    }

    match /tenant_control/{tenantId} {
      allow read, write: if isSuperAdmin();
    }

    match /tenant_access/{docId} {
      allow read, write: if isSuperAdmin();
    }

    match /superback_audit/{docId} {
      allow read, write: if isSuperAdmin();
    }
  }
}
