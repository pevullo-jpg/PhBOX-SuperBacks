// Va adattato quando il frontend farmacia sara tenant-aware.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }

    function isSuperuser() {
      return isSignedIn() &&
        exists(/databases/$(database)/documents/superback_config/main) &&
        request.auth.token.email != null;
    }

    match /superback_config/{docId} {
      allow read: if isSignedIn();
      allow write: if false;
    }

    match /tenants_public/{tenantId} {
      allow read, write: if false;
    }

    match /tenant_control/{tenantId} {
      allow read, write: if false;
    }

    match /tenant_access/{loginEmail} {
      allow read, write: if false;
    }

    match /superback_audit/{logId} {
      allow read, write: if false;
    }
  }
}
